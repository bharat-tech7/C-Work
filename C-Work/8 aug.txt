Q.2:Write a c program for finding maximum value number with 3 numbers?
sol:
#include<stdio.h>
int main()
{
    int a;
    int b;
    int c;
    printf("Enter first number :");
    scanf("%d",&a);
    printf("Enter second number :");
    scanf("%d",&b);
    printf("Enter third number");
    scanf("%d",&c);
    if(a<b&&b>c)
    {
        printf("b is greater ");
        
    }
    else if (a>b&&a>c)
    {
        printf("a is greater ");
    }
    else if (b<c&&c>a)
    {
        printf("c is greater");
    }
    return 0;
}
